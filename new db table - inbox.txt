CREATE TABLE IF NOT EXISTS `inbox` (
  `id` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `place_id` int(10) unsigned NOT NULL,
  `member_id` int(10) unsigned NOT NULL,
  `subject` text COLLATE utf8mb4_unicode_ci NOT NULL,
  `message` text COLLATE utf8mb4_unicode_ci NOT NULL,
  `parent_id` int(11) unsigned DEFAULT '0',
  `reply` tinyint(1) NOT NULL DEFAULT '0',
  `status` tinyint(1) NOT NULL DEFAULT '1',
  PRIMARY KEY (`id`),
  KEY `inbox_place_id_foreign` (`place_id`),
  KEY `inbox_member_id_foreign` (`member_id`),
  CONSTRAINT `inbox_member_id_foreign` FOREIGN KEY (`member_id`) REFERENCES `member` (`id`),
  CONSTRAINT `inbox_place_id_foreign` FOREIGN KEY (`place_id`) REFERENCES `place` (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=6 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
